Suitable on large screens 1280px width and above

NOTICE
--install this following dependencies, if you had them already installed is fine.

-Bootstrap library
npm install bootstrap 

- Boostrap icons
npm install bootstrap-icons

-Popper.js
npm install popper.js

-Maybe you can install jquery if youu like
-npm install jquery

---Copy and paste this in your index.js in the src folder

import "bootstrap/dist/css/bootstrap.min.css";
import "bootstrap/dist/js/bootstrap.bundle.min";
import $ from "jquery";
import Popper from "popper.js";
import "bootstrap-icons/font/bootstrap-icons.css";
